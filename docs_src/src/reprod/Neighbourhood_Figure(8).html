# Things we want to retain for creating a rank visualisation
box_metric_labels = []
box_vals = []
box_ranks = []
box_k = []
box_measure = []
box_dataset = []
box_dataset_type = [] # Discrete or cont
box_dataset_subset = [] # Rare/ultra rare etc
box_dataset_source = []
box_dataset_condition = []

measures = ['PSI']
k_test = ['3K_stats','10K_stats','30K_stats', '50K_stats','100K_stats']
fig_dir = '../Data/'
inc_metrics = all_metrics
for dataset in results_sets:
        for measure in measures:
            dat = results_sets[dataset][measure].reset_index().pivot(columns='K_value', index='metric', values=measure)
            # Give each K a rank
            for k in k_test:
                # Order the dataset by a specific k column
                col_ranks = dat.sort_values(by=[k], ascending=False)
                # Get the metrics which are now ordered with 0 being the highest and the last being the lowest
                metrics_ordered = col_ranks[k].index
                for i, metric_label in enumerate(metrics_ordered):
                    box_vals.append(col_ranks[k][i])
                    box_ranks.append(i) # Save the rank 
                    box_metric_labels.append(metric_label) # save the label
                    box_k.append(k)
                    box_measure.append(measure)
                    box_dataset.append(dataset)
                    box_dataset_type.append(dataset.split(' - ')[0]) # Separate out the dataset
                    box_dataset_subset.append(dataset.split(' - ')[1])
                    box_dataset_source.append(dataset.split(' - ')[2])
                    box_dataset_condition.append(dataset.split(' - ')[0] + ' ' + dataset.split(' - ')[1])
