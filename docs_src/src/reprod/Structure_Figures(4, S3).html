K_value = ['3K_stats', '10K_stats', '30K_stats', '50K_stats', '100K_stats']

evals = ['PSI'] #, 'ARI'

for data, key, key2 in zip(datapaths, results_sets.keys(), complete_results.keys()):
    infile = open(os.path.join(data_dir,data), 'rb')
    dat = pickle.load(infile)
    infile.close()
    reform = {(level1_key, level2_key, level3_key): values
              for level1_key, level2_dict in dat.items()
              for level2_key, level3_dict in level2_dict.items()
              for level3_key, values in level3_dict.items()}
    results = pd.DataFrame(reform).T
    names = ['Eval', 'metric', 'K_val']
    results.index.set_names(names, inplace=True)
    complete_results[key2] = results
    all_results = dict.fromkeys(K_value, )
    for K in K_value: 
        eval_df = pd.DataFrame(columns=new_metrics)#eval_df = pd.DataFrame(columns= metrics)
        for metric,n_metric in zip(metrics, eval_df.columns):#for metric in eval_df.columns:
            for eval in evals: # for K in K_value:
                series = results.loc[pd.IndexSlice[f'{eval}', f'{metric}', f'{K}']]
                series = series.fillna(-100)
                series = series[series.values > -100]
                eval_df.at[eval, n_metric] = series.mean()
        all_results[K] = eval_df

    final_df = pd.DataFrame.from_dict({(i, j): all_results[i][j]
                                for i in all_results.keys()
                                for j in all_results[i].keys()}, orient='index')
    
    names = ['K_value', 'metric']
    final_df.index.set_names(names, inplace=True)
    results_sets[key] = final_df

### BOXPLOT SET UP ###
# Things we want to retain for creating a rank visualisation
box_metric_labels = []
box_vals = []
box_ranks = []
box_k = []
box_measure = []
box_dataset = []
box_dataset_type = [] # Discrete or cont
box_dataset_subset = [] # Rare/ultra rare etc
box_dataset_condition = []

measures = ['PSI']
k_test = k_sub # this is ['3K_stats','10K_stats','30K_stats', '50K_stats']
fig_dir = '../Data/'
inc_metrics = all_metrics
save = False #True #
for dataset in results_sets:
        for measure in measures:
            dat = results_sets[dataset][measure].reset_index().pivot(columns='K_value', index='metric', values=measure)
            # Give each K a rank
            for k in k_test:
                # Order the dataset by a specific k column
                col_ranks = dat.sort_values(by=[k], ascending=False)
                # Get the metrics which are now ordered with 0 being the highest and the last being the lowest
                metrics_ordered = col_ranks[k].index
                for i, metric_label in enumerate(metrics_ordered):
                    box_vals.append(col_ranks[k][i])
                    box_ranks.append(i) # Save the rank 
                    box_metric_labels.append(metric_label) # save the label
                    box_k.append(k)
                    box_measure.append(measure)
                    box_dataset.append(dataset)
                    box_dataset_type.append(dataset.split(' - ')[0]) # Separate out the dataset
                    box_dataset_subset.append(dataset.split(' - ')[1])
                    box_dataset_condition.append(dataset.split(' - ')[0] + ' ' + dataset.split(' - ')[1])
